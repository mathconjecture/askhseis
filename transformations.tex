\begin{center}
    \minibox{\large \bfseries \textcolor{Col1}{Ασκήσεις στους Γραμμικούς
      Μετασχηματισμούς}}
\end{center}

\vspace{\baselineskip}

\begin{enumerate}
    \item Να δείξετε ότι οι παρακάτω μετασχηματισμοί είναι Γραμμικοί.
        \begin{enumerate}[i)]
            \item $ F(x,y) = (x+y,x) $
            \item $ F(x,y) = (2x+y,3x+2y) $
            \item $ F(x,y,z) = (x,0,z) $\
            \item $ F(x,y,z) = (x+y+z,2x-3y+4z) $
            \item $ F(x,y,z) = (x+2y-3z,4x-5y+6z) $
        \end{enumerate}

    \item Να δείξετε ότι οι παρακάτω μετασχηματισμοί δεν είναι Γραμμικοί.
        \begin{enumerate}[i)]
            \item $ F(x,y) = (x+1,y+2) $
            \item $ F(x,y) = (xy,x) $
            \item $ F(x,y) = (x^{2},y^{2}) $
            \item $ F(x,y) = (x+3,2y,x+y) $
        \end{enumerate}

    \item Αφού βρεθεί ο πίνακας των παρακάτω γραμμικών μετασχηματισμών να βρεθεί μια βάση 
      και η διάσταση της εικόνας και του πηρήνα τους.
        \begin{enumerate}[i)]
            \item $ F(x,y,z) = (x+2y-z,y+z,x+y-2z) $
            \item $ F(x,y,z) = (x+2y-3z,2x+5y-4z,x+4y+z) $
            \item $ F(x,y,z) = (x+y,y+z) $
            \item $ F(x,y,z) = (x+y+z,2x+2y+2z) $
            \item $ F(x,y,z,w) = (x-y+z+w,x+2z-w,x+y+3z-3w) $
            \item $ F(x,y,z,w) = (x-y+z+w,2x-2y+3z+4w,3x-3y+4z+5w) $
            \item $ F(x,y,z,w) = (x+2y+3z+2w,2x+4y+7z+5w,x+2y+6z+5w) $
            \item $ F(x,y,z,u,w) = (x+2y+2z+u+w,x+2y+3z+2u-w,3x+6y+8z+5u-w) $
        \end{enumerate}

    \item Να βρεθεί μια βάση και η διάσταση της εικόνας και του πυρήνα των παρακάτω γραμμικών
        μετασχηματισμών.

        \begin{enumerate*}[i)]
        \item $\begin{pmatrix*}[r]
                1 & 2 & 3 & 1 \\
                1 & 3 & 5 & -2 \\
                3 & 8 & 13 & -3
            \end{pmatrix*}$ 
            \quad 
        \item $\begin{pmatrix*}[r]
                1 & 2 & 0 & 1 \\
                2 & -1 & 2 & -1 \\
                1 & -3 & 2 & -2
            \end{pmatrix*}$
            \quad
        \item $ \begin{pmatrix*}[r]
                1 & 0 & 2 & -1 \\
                2 & 3 & -1 & 1 \\
                -2 & 0 & -5 & 3 
            \end{pmatrix*} $
        \end{enumerate*}


\end{enumerate}
