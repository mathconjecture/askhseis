\begin{center}
  \fbox{\Large\bfseries Ασκήσεις ΣΔΕ $1$ης Τάξης}
\end{center}

\vspace{\baselineskip}

\begin{enumerate}
  \item Να λυθούν οι παρακάτω σδε 1ης τάξης.

\begin{description}
  \setlength{\itemsep}{\baselineskip}
  \item[(Θέμα Ιουν 2018)] $(y+xy^{2})dx-xdy=0, \quad y(1)=2$. {(\bfseries Γραφ. Παρ.)}
  \hfill Απ: $y(x)=\frac{2-x^{2}}{2x}$
  \item[(Θέμα Ιουν 2018)] $y=(1+x){y'}^{2}$ \hfill Απ: $x(t)=\frac{c_{1}}{(t-1)^{2}}-1$, $y(t)=\frac{c_{1}}{(t-1)^{2}t^{2}}$
  \item[(Θέμα Ιουν 2018)] $xy'-4y=x^{2}\sqrt{y}$, \quad $y(1)=1$,\; $x>0$ \hfill Απ: $y(x)=x^{2}\left(1+\frac{\ln x}{2}\right)$
  \item[(Θέμα Ιουν 2017)] $y'=\frac{e^{x}}{(1+e^{x})y}$ {\bfseries (2 ολ. παραγ.)}

  \hfill Απ: $\mu(x)=\frac{1}{1+e^{x}}$, $\mu(y)=e^{-\frac{y^{2}}{2}}$, $\ln(1+e^{x})-\frac{y^{2}}{2}=c$
  \item[(Θέμα Σεπ 2017)] $y'=-\frac{3xy+y^{2}}{x^{2}+xy}$ \hfill Απ: $\mu(x)=\frac{1}{x}$, $x^{3}y+\frac{x^{2}}{2}y^{2}=c$
  \item[(Θέμα Ιουν 2016)] $2xy^{3}dx+(3x^{2}y^{2}+x^{2}y^{3}+1)dy=0$ \hfill Απ: $\mu(y)=e^{y}$, $x^{2}y^{3}e^{y}+e^{y}=c$
  \item[(Θέμα Ιουν 2016)] $y'=\frac{x+y}{x-y}$ \hfill Απ: $\arctan(\frac{y}{x})-\frac{1}{2}\ln\left(1+\Bigl(\frac{y}{x}\Bigr)^{2}\right)=\ln\abs{x}+c$
  \item[(Θέμα Σεπ 2016)] $y'-\frac{1}{3x}y=\frac{x}{3y}$ \hfill Απ: $y^{2}(x)=cx^{3}-\frac{x^{2}}{3}$
\end{description}

\end{enumerate}
