\renewcommand{\theenumii}{\roman{enumii}}
\begin{enumerate}

\item Να υπολογιστούν τα παρακάτω όρια κάνοντας άρση της απροσδιοριστίας, 
    με αλγεβρικό τρόπο.

\begin{enumerate}

\item $\lim_{x\to -2}\frac{-3x+6}{x^2-2x}$ \hfill Απ: $\frac{3}{2}$
\item $\lim_{x\to 0}\frac{5x^{3}+8x^{2}}{3x^{4}-16x^{2}}$ \hfill Απ: $-\frac{1}{2}$
\item $\lim_{x\to 2}\frac{x^{2}-5x+6}{x-2}$ \hfill Απ: $-1$
\item $\lim_{x\to -5}\frac{x^{2}+3x-10}{x+5}$ \hfill Απ: $-7$
\item $\lim_{x\to 1}\frac{x^{2}+x-2}{x^{2}-x}$ \hfill Απ: $3$
\item $\lim_{x\to 1}\frac{\frac{1}{x-1}+ \frac{1}{x+1}}{x}$ \hfill Απ: $-2$
\item $\lim_{x\to 1}\frac{x^4-1}{x^3-1}$ \hfill Απ: $\frac{4}{3}$
\item $\lim_{x\to 16}\frac{\sqrt{x}-4}{x-16}$ \hfill Απ: $\frac{1}{8}$
\item $\lim_{x\to 1}\frac{x-1}{\sqrt{x+3} -2}$ \hfill Απ: $4$
\item $\lim_{x\to -1}\frac{\sqrt{x^{2}+8} -3}{x+1}$ \hfill Απ: $- \frac{1}{3} $


\end{enumerate}


\item Να υπολογιστούν τα παρακάτω όρια με τη βοήθεια του ορίου $\lim\limits_{x\to 0}\frac{\sin x}{x}=1$.


\begin{enumerate}

    \item $\lim_{x\to 0}\frac{\sin{2x}}{4x}$ \hfill Απ: $ \frac{1}{2}$
\item $\lim_{x\to 0}\frac{\sin{7x}}{x}$ \hfill Απ: $7$
\item $\lim_{x\to 0}\frac{x}{\sin 3x}$ \hfill Απ: $\frac{1}{3}$
\item $\lim_{x\to 0}\frac{2x}{\tan x}$ \hfill Απ: $2$
\item $\lim_{x\to 0}\frac{\tan{x}}{3x\cdot \cos{2x}}$ \hfill Απ: $ \frac{1}{3}$
\item $\lim_{x\to 0}\frac{x^{2}-x + \sin{x}}{2x}$ \hfill Απ: $0$
\item $\lim_{x\to 0}\frac{\sin{5x}}{\sin{4x}}$ \hfill Απ: $ \frac{5}{4} $

\end{enumerate}

\newpage

\item Να υπολογιστούν τα παρακάτω όρια στο άπειρο με χρήση του κριτηρίου Παρεμβολής.

\begin{enumerate}

\item $\lim_{x\to \pm\infty}\frac{\sin x}{x}$ \hfill Απ: $0$
\item $\lim_{x\to \infty}\frac{\sin 2x}{x}$ \hfill Απ: $0$
\item $\lim_{x\to -\infty}\frac{\cos x}{3x}$ \hfill Απ: $0$
\item $\lim_{x\to 0}x^4\cos(\frac{2}{x^{2}})$ \hfill Απ: $0$

\end{enumerate}

\item Να υπολογιστούν τα όρια των παρακάτω συναρτήσεων στο άπειρο.

\begin{enumerate}



\item $\lim_{x\to \infty}\frac{x+1}{x^2+3}$ \hfill Απ: $0$
\item $\lim_{x\to \infty}\frac{2x^3+7}{x^3-x^2+x+7}$ \hfill Απ: $2$
\item $\lim_{x\to \infty}\sqrt{\frac{8x^2-3}{2x^2 +3}}$ \hfill Απ: $2$
\item $\lim_{x\to -\infty}\left(\frac{1-x^3}{x^2-7x}\right)^5$ \hfill Απ: $\infty$
\item $\lim_{x\to \infty}\frac{2-x+\sin x}{1+\cos x}$ \hfill $-1$

\end{enumerate}

\item Να υπολογιστούν τα παρακάτω όρια, εξετάζοντας το \textbf{πρόσημο του 
    παρονομαστή}, κοντά στο $ x_{0} $.

\begin{enumerate}

\item $\lim_{x\to 0^+}\frac{1}{3x}$ \hfill Απ: $\infty$
\item $\lim_{x\to 0^-}\frac{5}{2x}$ \hfill Απ: $-\infty$
\item $\lim_{x\to 2^-}\frac{3}{x-2}$ \hfill Απ: $-\infty$
\item $\lim_{x\to -5^-}\frac{3x}{2x+10}$ \hfill Απ: $\infty$
\item $\lim_{x\to -8^+}\frac{2x}{x+8}$ \hfill Απ: $-\infty$
\item $\lim_{x\to 7}\frac{4}{(x-7)^2}$ \hfill Απ: $\infty$
\item $\lim_{x\to 0}\frac{-1}{x^2(x+1)}$ \hfill Απ: $-\infty$
\item $\lim_{x\to 0^+}\frac{2}{3x^{\frac{1}{3}}}$ \hfill Απ: $\infty$
\item $\lim_{x\to 0^-}\frac{2}{x^{\frac{1}{5}}}$ \hfill Απ: $-\infty$
\item $\lim_{x\to \left(\frac{\pi}{2}\right)^-}\tan x$ \hfill Απ: $\infty$
\item $\lim_{x\to 2^+}\frac{1}{x^2-4}$ \hfill Απ: $\infty$
\item $\lim_{x\to 2^-}\frac{1}{x^2-4}$ \hfill Απ: $-\infty$
\item $\lim_{x\to -2^+}\frac{1}{x^2-4}$ \hfill Απ: $-\infty$
\item $\lim_{x\to -2^-}\frac{1}{x^2-4}$ \hfill Απ: $\infty$
\item $\lim_{x\to -2^+}\frac{x^2-1}{2x+4}$ \hfill Απ: $\infty$
\item $\lim_{x\to -2^-}\frac{x^2-1}{2x+4}$ \hfill Απ: $-\infty$
\end{enumerate}

\item Να υπολογιστούν τα όρια των παρακάτω \textbf{ρητών} συναρτήσεων.

\begin{enumerate}
\item $\lim_{x\to \infty}\frac{1}{3x+4}$ \hfill Απ: $0$
\item $\lim_{x\to -\infty}\frac{8x^4+3x-5}{x^2+1}$ \hfill Απ: $\infty$
\item $\lim_{x\to \infty}\frac{3x+4}{x-7}$ \hfill Απ: $3$
\item $\lim_{x\to 0}\frac{x^{2}-4x+4}{x^3+5x^2-14x}$ \hfill Απ: δεν υπάρχει
\item $\lim_{x\to -\infty}x^2(x+1)(3-x)$ \hfill Απ: $-\infty$
\item $\lim_{x\to \infty}\cos x$ \hfill Απ: δεν υπάρχει
\item $\lim_{x\to \infty}\frac{5x^4+5}{(x^2-1)(2x^3+3)}$ \hfill Απ: $0$
\end{enumerate}

\item Να υπολογιστούν τα παρακάτω όρια κάνοντας κάνοντας \textbf{άρση της απροσδιοριστίας} και χρήση του κανόνα L' Hospital όπου χρειάζεται.

\begin{enumerate}
\item $\lim_{x\to \infty}\frac{\ln x}{x^2}$ \hfill Απ: $0$
\item $\lim_{x\to (\frac{\pi}{2})^-}\frac{\cos x}{1-\sin x}$ \hfill Απ: $\infty$
\item $\lim_{x\to 0^+}\frac{\ln(e^x-1)}{\ln x}$ \hfill Απ: $1$
\item $\lim_{x\to 0}\frac{1-\cos x }{\tan x}$ \hfill Απ: $0$
\item $\lim_{x\to 0}\frac{\sin 4x}{2x}$ \hfill Απ: $2$
\item $\lim_{x\to \infty}x^2e^{-x}$ \hfill Απ: $0$
\item $\lim_{x\to 0^+}x\ln x$ \hfill Απ: $0$
\item $\lim_{x\to \infty}x\sin(\frac{1}{x})$ \hfill Απ: $1$
\item $\lim_{x\to \infty}(\sqrt{x} - x)$ \hfill Απ: $-\infty$
\item $\lim_{x\to \infty}x^{\frac{1}{x^2}}$ \hfill Απ: $1$
\item $\lim_{x\to 0^+}(\cos x)^{\frac{2}{x}}$ \hfill Απ: $1$


\end{enumerate}

\end{enumerate}
