\begin{center}
  \fbox{\Large\bfseries Θέματα ΣΔΕ $2$ης και ανώτερης Τάξης}
\end{center}

\vspace{\baselineskip}

\begin{enumerate}
  \item Να λυθούν οι παρακάτω σδε 1ης τάξης.

\begin{description}
  \setlength{\itemsep}{\baselineskip}
  \item[(Θέμα Ιουν 2018)] $y'' + y = 4 e^{x} + 2\sin x$\quad (Προσδ. συντελ.)
  \item[(Θέμα Σεπ 2016)] $x^{2}y''+xy'-4y=x^{2}-\ln x$, $x>0$\quad (\textlatin{Euler})
  \item[(Θέμα Σεπ 2016)] $y''-4y'+4y=e^{x}+1+\sin x$\quad (Προσδ. συντελ.)
  \item[(Θέμα Ιουν 2017)] Για τη γραμμική σδε $y^{(3)}+\frac{1}{x^{2}}y'-\frac{1}{x^{3}}y=r(x)$, $x>0$
  \begin{enumerate}[i)]
    \item Εξετάστε αν οι λύσεις της ομογενούς $y_{1}=x$, $y_{2}=x\ln x$, $y_{3}=x\ln^{2}x$, αποτελούν θεμελιώδες σύνολο λύσεων.
    \item Αν $r(x)=\frac{2}{x}$, βρείτε την γενική λύση της εξίσωσης.
  \end{enumerate}
  \item[(Θέμα Ιουν 2017)] $yy''+2y'=2(y')^{2}, y(0)=1, y'(0)=2$
  \item[(Θέμα Ιουν 2017)] $x^{2}y''+xy'+4y=\ln x- \frac{1}{x^{2}}$\quad (\textlatin{Euler}, Προσδ. συντελ.)
\end{description}

\end{enumerate}
