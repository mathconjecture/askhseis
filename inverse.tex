\begin{enumerate}

\item Να υπολογισθεί ο αντίστροφος των παρακάτω $2\times 2$ πινάκων.

\begin{enumerate}[i)]

\item $A=\begin{pmatrix}
  4 & 3 \\
  1 & 1
\end{pmatrix}$\hfill Απ: $Α^{-1}=\begin{pmatrix}
\phantom{-}1 & -3 \\
-1 & \phantom{-}4
\end{pmatrix}$

\item $B=\begin{pmatrix}
  4 & 3 \\
  3 & 2
\end{pmatrix}$\hfill Απ: $B^{-1}=\begin{pmatrix}
-2 & \phantom{-}3 \\
\phantom{-}3 & -4
\end{pmatrix}$

\item $C=\begin{pmatrix}
  1 & 1 \\
  5 & 4
\end{pmatrix}$\hfill Απ: $C^{-1}=\begin{pmatrix}
-4 & \phantom{-}1 \\
\phantom{-}3 & -1
\end{pmatrix}$

\item $D=\begin{pmatrix}
  3 & 1 \\
  4 & 2
\end{pmatrix}$\hfill Απ: $D^{-1}=\begin{pmatrix}
\phantom{-}1 & -\sfrac{1}{2} \\
-2 & \phantom{-}\sfrac{3}{2}
\end{pmatrix}$

\end{enumerate}

\item Να υπολογισθεί ο αντίστροφος των παρακάτω $3\times 3$ πινάκων.

\begin{enumerate}[i)]

  \item $A=\begin{pmatrix}
      \phantom{-}1 & -1 &   \phantom{-}1\\
      \phantom{-}0 & -2 &   \phantom{-}1 \\

   \end{pmatrix}$\hfill Απ: $A^{-1}=\begin{pmatrix}
    \phantom{-}3 & -3 & \phantom{-}1 \\
    -2 & \phantom{-}2 & -1 \\
    -4 & \phantom{-}5  & -2
    \end{pmatrix}$

    \item $B=\begin{pmatrix}
        1 & 3 & 3 \\
        1 & 4 & 3 \\
        1 & 3 & 4
      \end{pmatrix}$\hfill Απ: $B^{-1}=\begin{pmatrix}
      \phantom{-}7 & -3 & -3 \\
      -1 & \phantom{-}1 & \phantom{-}0 \\
      -1 & \phantom{-}0  & \phantom{-}1
      \end{pmatrix}$

      \item $C=\begin{pmatrix}
          \phantom{-}7 & \phantom{-}2 & \phantom{-}1\\
          \phantom{-}0 & \phantom{-}3 & -1 \\
         -3 & \phantom{-}4 & -2
       \end{pmatrix}$\hfill Απ: $C^{-1}=\begin{pmatrix}
        -2 & \phantom{-}8 & -5 \\
        \phantom{-}3 & -11 & \phantom{-}7 \\
        \phantom{-}9 & -34  & \phantom{-}21

\item $D=\begin{pmatrix}
    1 & 2 & 3\\
    0 & 1 & 4 \\

  \end{pmatrix}$\hfill Απ: $D^{-1}=\begin{pmatrix}
  -24 & \phantom{-}18 & \phantom{-}5 \\
  \phantom{-}20 & -15 & -4 \\
  -5 & \phantom{-}4  & 1
  \end{pmatrix}$

\item  $E = \begin{pmatrix*}
        7 & 0 & 2 \\
        4 & 1 & 1 \\
        3 & 0 & 1
    \end{pmatrix*}$ 
    \hfill Απ: $ E^{-1} = \begin{pmatrix*}[r]
       1 & 0 & -2 \\
       -1 & 1 & 1 \\
       -3 & 0 & 7
   \end{pmatrix*} $ 


\end{enumerate}



\end{enumerate}
