  \item Να υπολογιστούν τα παρακάτω όρια:
    \begin{enumerate}[i)]
      \item $ \lim\limits_{(x,y)\to (\pi, \pi)} \frac{1}{xy} \sin{(x+y)} $ 
        \begin{proof}
          \[
            \lim\limits_{(x,y)\to (\pi, \pi)} \frac{1}{xy} \sin{(x+y)} = 
            \frac{1}{\pi ^{2}} \sin{2 \pi} = 0 
          \]    
        \end{proof}

      \item $ \lim\limits_{(x,y)\to (0, 0)} \sqrt{x^{2}+y^{2}} 
        \cos{\frac{1}{\sqrt{xy}}} $
        \begin{proof}
        \item {}
          \begin{minipage}{0.50\textwidth}
            \begin{itemize}
              \item $ \abs{\sqrt{x^{2}+y^{2}} \cos{\frac{1}{\sqrt{xy}}}} = 
                \abs{\sqrt{x^{2}+y^{2}}} \cdot 
                \abs{\cos{\frac{1}{\sqrt{xy}}}} \leq 
                \abs{\sqrt{x^{2}+y^{2}}} $ \hfill \tikzmark{a}
              \item $ \lim\limits_{(x,y)\to (0, 0)} \sqrt{x^{2}+y^{2}} = 0 $ 
                \hfill \tikzmark{b}
            \end{itemize}
          \end{minipage}   
          \mybrace{a}{b}[$ \lim\limits_{(x,y)\to (0, 0)} 
          \sqrt{x^{2}+y^{2}} \cos{\frac{1}{\sqrt{xy}}} = 0 $]
        \end{proof}
    \end{enumerate}
\end{enumerate}
