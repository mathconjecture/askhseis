\section*{Γραμμικά Συστήματα}

\begin{enumerate}

  \item Να λυθούν τα παρακάτω γραμμικά συστήματα $ 2 \times 2 $ με αντικατάσταση και με 
    μέθοδο αντίθετων συντελεστών.

    \begin{enumerate}[i)]
      \item $ 
      \sysdelim.\}
      \systeme{
        x + 2y = 5, 
        4x + y = 6
      } $ 
      \hfill Απ: 
      \begin{tabular}{l}  
        $x=1 $ \\ 
        $y=2 $ 
      \end{tabular}

    \item $ 
    \sysdelim.\}
    \systeme{
      4x+3y=11,
      5x+7y=17
    } $ 
    \hfill Απ: 
    \begin{tabular}{l}  
      $x=2 $ \\ 
      $y=1 $ 
    \end{tabular}
\end{enumerate}

\end{enumerate}
